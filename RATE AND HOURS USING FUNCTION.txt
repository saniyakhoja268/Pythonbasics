def computepay(h,r):
    if (0<h and h<40):                  
        pay=r*h
        print(f"{pay}")
    elif (40<h and h<60):
        pay=(r*40)+(h-40)*(r*1.5)
        print(f"{pay}")
    elif (60<h):
        pay=(r*40)+(20*r*1.5)+(h-60)*(r*2.0)
        print(f"{pay}")
    else:
        print("Invalid")

h= int(input("Enter hours:"))
r= int(input("Enter rate:"))
